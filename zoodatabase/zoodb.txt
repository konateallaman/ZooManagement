    DECLARE age INT;
    SET age = TIMESTAMPDIFF(YEAR, NEW.CustDOB, CURDATE());

        SET MESSAGE_TEXT = 'Failed: Visitor must be at least 18 years old.';
    END IF;
END
$$

    END IF
$$
